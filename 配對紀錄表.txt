CREATE TABLE MATCHES (
    MATCH_ID INT IDENTITY(2001,1) NOT NULL PRIMARY KEY, 
    USER1_ID INT NOT NULL, 
    USER2_ID INT NOT NULL, 
    MATCH_DATE DATETIME NOT NULL, 配對日期時間
    MATCH_SUCCESS BIT NOT NULL DEFAULT 0,  配對是否成功的標誌位，預設為未成功配對
   
    CHECK (USER1_ID < USER2_ID) -- 確保 USER1_ID 小於 USER2_ID，以確保每個配對都是唯一的
);
